# array test: sieve of Eratosthenes gets all prime numbers


sub primes( max )

  mark = []
  ret = []

  n = 2 
  while n < max
    m = 2 * n
    loop 
        if m > max
            break
        end
        mark[ m ] = 1
        m = m + n
    end
    n = n + 1
  end

  n = 2
  while n  < max
    if mark[ n ] != 1
       push( ~array ret ~top n )
    end
    n = n + 1
  end

  return ret
end
