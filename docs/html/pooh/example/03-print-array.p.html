# array test: print all elements of array, iterate with while statement.

arr = [  1, 2, 3, 5, 7, 11, 13, 17 ]

i = 1
while arr [ i ] != Null
    println( ~msg 'prime number i ' .. arr[ i]  )
    i = i + 1
end
